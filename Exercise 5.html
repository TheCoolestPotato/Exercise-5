ci_quant <- function(x, level = 0.95){
  ci <- quantile(x, c((1-level)/2, 1 - (1-level)/2)) |> round(3)
  ci <- paste0("[", ci[1], "-", ci[2], "]")
  names(ci) <- "CI"
  return(ci)}

yarr_booty <- function(x){
  boot <- NULL
  for (i in 1:1000) {
    boot[i] <- mean(sample(x, length(x), replace = TRUE))
  }
  ci <- quantile(boot, c(.05/2, 1-.05/2))
}
